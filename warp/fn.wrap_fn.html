Combines received filter with pre and after filters

Example

use crate::warp::Filter;

let route = warp::any()
    .map(|| "hello world")
    .with(warp::wrap_fn(|filter| filter));

You can find the full example in the usage example.

fn main() {
    // Match any request and return hello world!
    let routes = warp::any()
        .map(|| "hello world")
        .boxed()
        .recover(|_err| async { Ok("recovered") })
        // wrap the filter with hello_wrapper
        .with(warp::wrap_fn(hello_wrapper));

    warp::serve(routes).run(([127, 0, 0, 1], 3030)).await;
}
